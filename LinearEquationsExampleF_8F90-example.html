4PROGRAM linear_equations_example
5 USE iterative_solver
6 IMPLICIT NONE
7 INTEGER, PARAMETER :: n = 30, nroot = 2
8 DOUBLE PRECISION, PARAMETER :: alpha = 300
9 DOUBLE PRECISION, DIMENSION(5), PARAMETER :: augmented_hessian_factors = [0.0_8, .001_8, .01_8, .1_8, 1.0_8]
10 DOUBLE PRECISION, DIMENSION (n, n) :: m
11 DOUBLE PRECISION, DIMENSION (n, nroot) :: c, g, rhs
12 DOUBLE PRECISION, DIMENSION (nroot) :: error
13 DOUBLE PRECISION :: augmented_hessian
14 INTEGER :: i, j, root, iaug, nwork
15 LOGICAL :: converged
16 print *, 'Fortran binding of IterativeSolver'
17 call mpi_init
18 DO i = 1, n; m(i, i) = alpha * i + 2 * i - 2; DO j = 1, n; IF (i.NE.j) m(i, j) = i + j - 2;
19 END do;
20 END DO
21 DO i = 1, nroot; DO j = 1, n; rhs(j, i) = 1 / dble(j + i - 1);
22 END do;
23 END DO
24 nwork = nroot
25 DO iaug = 1, SIZE(augmented_hessian_factors)
26 augmented_hessian = augmented_hessian_factors(iaug)
27 print *, 'solve linear system with augmented hessian factor ', augmented_hessian
28 CALL iterative_solver_linear_equations_initialize(n, nroot, rhs, augmented_hessian, thresh = 1d-11, verbosity = 1)
29 c = 0; DO i = 1, nroot; c(i, i) = 1;
30 ENDDO
31 DO i = 1, n
32 g = matmul(m, c)
33 nwork = iterative_solver_add_vector(c, g)
34 DO root = 1, nwork
35 DO j = 1, n
36 c(j, root) = c(j, root) - g(j, root) / m(j, j)
37 END DO
38 END DO
39 nwork = iterative_solver_end_iteration(c, g)
40 IF (nwork .LE. 0) EXIT
41 END DO
42 print *, 'error =', iterative_solver_errors()
43 DO i = 1, nroot
44 print *, 'solution ', c(1 : min(n, 10), i)
45 END DO
46 Call iterative_solver_print_statistics
47 CALL iterative_solver_finalize
48 ENDDO
49 CALL mpi_finalize
50END PROGRAM linear_equations_example
